# 常用 SQL 命令：
\dt                          # 查看所有表
SELECT * FROM assets;        # 查看资产数据
SELECT * FROM ports;         # 查看端口数据
SELECT COUNT(*) FROM assets; # 统计资产数量
\q                           # 退出
-- 查看所有资产及其端口
SELECT a.domain, a.url, a.status_code, p.port, p.service, p.version 
FROM assets a 
LEFT JOIN ports p ON a.id = p.asset_id;

-- 查看开放端口统计
SELECT port, service, COUNT(*) as count 
FROM ports 
GROUP BY port, service 
ORDER BY count DESC;

-- 查看最近发现的资产
SELECT * FROM assets ORDER BY created_at DESC LIMIT 10;
